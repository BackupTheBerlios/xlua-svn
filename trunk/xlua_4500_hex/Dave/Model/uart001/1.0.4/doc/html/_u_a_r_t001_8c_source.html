00291 /* Give optimized PDIV and STEP value for the given baud rate */ 
00292 void UART001_lConfigureBaudRate\
00293                               (uint32_t BaudRate,uint32_t* Pdiv,uint32_t* Step)
00294 {
00295         float ratio = (float)0;
00296         float TempBaud = (float)0;
00297         float TempPdiv = (float)0;
00298         float TempStep = (float)0;
00299         TempBaud = (float)BaudRate;
00300         ratio = (((((float)UART001_CLOCK * (float)1000000)/ \
00301              ((float)UART001_DCTQ_VALUE * (float)UART001_MAX_STEPVALUE)) * \
00302             UART001_IM_STEPVALUE)/TempBaud);
00303         /* Calculating optimum PDIV value */
00304         if( ratio <= (float)1)
00305         {
00306                 TempPdiv = (float)0;
00307         }
00308         else
00309         {
00310                 TempPdiv =  ratio - (float)1;
00311                 if( TempPdiv > UART001_MAX_VALUE)
00312                 {
00313                         TempPdiv = UART001_MAX_VALUE;
00314                 }
00315         }
00316         *Pdiv = (uint32_t)TempPdiv;
00317         /* Calculating STEP value */
00318         TempStep = ((((TempBaud * (float)UART001_DCTQ_VALUE * \
00319                      (float)UART001_MAX_STEPVALUE) / (float)1000000) * \
00320                      ((float)1+ (float)(*Pdiv))) / (float)(UART001_CLOCK));
00321 
00322         if(TempStep > UART001_MAX_VALUE)
00323         {
00324                 TempStep = UART001_MAX_VALUE;
00325         }
00326         *Step = (uint32_t)TempStep;
00327 }

00384                  RD_REG(UartRegs->PSR_ASCMode,USIC_CH_PSR_ASCMode_RXIDLE_Msk, \
